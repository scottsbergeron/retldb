106 }
107 }
108
109 // Register the type
110 g_type_registry[g_type_count].id = id;
111 g_type_registry[g_type_count].name = name;
112 g_type_registry[g_type_count].size = size;
113 g_type_registry[g_type_count].compare = compare;
114 g_type_registry[g_type_count].copy = copy;
115 g_type_registry[g_type_count].free = free;
116 g_type_registry[g_type_count].serialize = serialize;
117 g_type_registry[g_type_count].deserialize = deserialize;
118
119 g_type_count++;
